<script>

  const API_KEY = "gDaxorrfRitBH5xZg9Zusd4d2rfpEfXudILuJxMa3270600d";

  const canvas = document.getElementById("diceCanvas");


  let dice;


  function initDice() {

    dice = new dddice.ThreeDDice(canvas, API_KEY, {

      width: window.innerWidth,

      height: window.innerHeight,

      controlsEnabled: false

    });


    dice.start();

  }


  document.getElementById("rollButton").onclick = async () => {

    if (!dice) return;


    try {

      const result = await dice.roll([{ type: "d20", count: 1 }]);

      console.log("RESULT:", result);


      // Optional: Send result back to Framer

      if (parent) {

        parent.postMessage({ type: "d20_result", value: result }, "*");

      }

    } catch (err) {

      console.error("ROLL FAILED:", err);

    }

  };


  initDice();

</script>
